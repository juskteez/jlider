Slider #2 Custom Usage

    $("#jl2").jliderSlide({
        slideEffect  : "random",
        visiTime     : 6000,
        autoPlay     : true,
        width        : 600,
        height       : 320,
        responsive   : false,
        photoCaption : true,
        noCopy       : true,
        hideControl  : "hover",
        fullScreen   : true,
        navType      : "line",
        Loaded       : function (curSlide) {
                        	console.log('Loaded...!');
                    	},
        pressPlay    : function (curSlide) {
                            console.log('Pressed Play button...!');
                        },
        pressPause   : function (curSlide) {
                            console.log('Pressed Pause button...!');
                        }
    });
